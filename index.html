            // 解析题目，提取选项

            function parseQuestion(text) {

                const lines = text.split('\n');

                const questionText = lines[0];

                

                // 提取选项

                const optionsLine = lines.find(line => line.includes('A.'));

                const options = [];

                

                if (optionsLine) {

                    const optionMatches = optionsLine.match(/[A-D]\.\s*\w+/g);

                    if (optionMatches) {

                        optionMatches.forEach(match => {

                            const [label, word] = match.split('.');

                            options.push({

                                label: label.trim(),

                                word: word.trim()

                            });

                        });

                    }

                }

                

                return {

                    questionText,

                    options

                };

            }

            

            // 显示选项

            function displayOptions(options) {

                optionsContainer.innerHTML = '';

                    const optionElement = document.createElement('div');

                    optionElement.className = 'option';

                    optionElement.innerHTML = `<span class="option-label">${option.label}.</span> ${option.word}`;

                    optionsContainer.appendChild(optionElement);

                });

            }

            

            // 高亮显示文本，并添加法语助手链接

            function highlightText(text) {

                const words = text.split(/\s+/);

                const highlightedWords = words.map(word => {

                    // 清理单词（去除标点符号）

                    const cleanWord = word.replace(/[.,;:!?()]/g, '');

                    if (cleanWord.length > 1) {

                        // 保留原始单词的标点符号

                        const punctuation = word.replace(cleanWord, '');

                        return `<span class="word" data-word="${cleanWord}">

                                    ${cleanWord}${punctuation}

                                    <span class="tooltip">点击查询 "${cleanWord}"</span>

                                </span>`;

                    }

                    return word;

                });

                

                highlightedText.innerHTML = highlightedWords.join(' ');

                

                // 为每个单词添加点击事件

                document.querySelectorAll('.word').forEach(wordElement => {

                    wordElement.addEventListener('click', function() {

                        const word = this.getAttribute('data-word');

                        window.open(`https://www.frdic.com/dicts/fr/${encodeURIComponent(word)}`, '_blank');

                    });

                });

            }

                    const verb = option.word;

                    const verbData = verbDatabase[verb] || {

                        meaning: '暂无数据',

                        prepositions: [],

                        usage: '暂无数据',

                        verbData.prepositions.forEach(prep => {

                            prepositionsHTML += `<p><strong>${prep.prep}</strong>: ${prep.example}</p>`;

                        });

                    } else {

                        prepositionsHTML = '<p>暂无介词搭配数据</p>';

                    }

                    

                    let examplesHTML = '';

                    if (verbData.examples && verbData.examples.length > 0) {

                        verbData.examples.forEach(example => {

                            examplesHTML += `<div class="usage-example">${example}</div>`;

                        });

                    }

                    

                    card.innerHTML = `

                        <h3>${option.label}. ${verb}</h3>

                        <div class="analysis-item">

                            <h4>核心意思</h4>

                            <p>${verbData.meaning}</p>

                        </div>

                        <div class="analysis-item">

                            <h4>常见介词搭配</h4>

                            ${prepositionsHTML}

                        </div>

                        <div class="analysis-item">

                            <h4>用法</h4>

                            <p>${verbData.usage}</p>

                        </div>

                        <div class="analysis-item">

                            <h4>例句</h4>

                            ${examplesHTML}

                        </div>

                    `;

                    

                    analysisContainer.appendChild(card);

                });

            }

                const analysis = `

                    <p>原句中"cherchent à favoriser"表示"寻求促进/有利于"，其中：</p>

                    <ul>

                        <li>"chercher à + 动词不定式"是一个常见结构，表示"寻求做某事"</li>

                        <li>"favoriser"作为及物动词，表示"促进，有利于"，后接直接宾语</li>

                    <ul>

                        <li><strong>rassurer（使放心，安慰）</strong>：语义上与原句不符，城市不是要"使人放心"生态出行方式</li>
